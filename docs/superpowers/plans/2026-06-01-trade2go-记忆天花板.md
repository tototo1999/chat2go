```python
# worker/trade_memfs.py
"""Anthropic memory tool 的 Supabase 后端:六命令 view/create/str_replace/insert/delete/rename。
Store 抽象出读写,便于单测注入 fake。命令结构以 Task 0 校准为准。"""
from __future__ import annotations


class SupabaseStore:
    """tradego_memory_files 后端,按 (expert_id, product, path) 寻址。"""
    def __init__(self, sb, expert_id: str, product: str = "tradego"):
        self.sb, self.expert_id, self.product = sb, expert_id, product

    def read(self, path):
        r = self.sb.table("tradego_memory_files").select("content") \
            .eq("expert_id", self.expert_id).eq("product", self.product) \
            .eq("path", path).maybe_single().execute()
        return r.data["content"] if r.data else None

    def write(self, path, content):
        self.sb.table("tradego_memory_files").upsert(
            {"expert_id": self.expert_id, "product": self.product,
             "path": path, "content": content, "updated_at": "now()"},
            on_conflict="expert_id,product,path").execute()

    def delete(self, path):
        self.sb.table("tradego_memory_files").delete() \
            .eq("expert_id", self.expert_id).eq("product", self.product).eq("path", path).execute()

    def list(self, prefix="/"):
        r = self.sb.table("tradego_memory_files").select("path") \
            .eq("expert_id", self.expert_id).eq("product", self.product) \
            .like("path", prefix + "%").execute()
        return sorted(x["path"] for x in (r.data or []))


def dispatch(store, tin: dict) -> dict:
    cmd = tin.get("command")
    path = tin.get("path", "")
    if cmd == "view":
        listed = store.list(path.rstrip("/") + "/")
        if listed and store.read(path) is None:          # 目录
            return {"files": listed}
        return {"content": store.read(path) or ""}
    if cmd == "create":
        store.write(path, tin.get("file_text", ""))
        return {"ok": True}
    if cmd == "str_replace":
        cur = store.read(path) or ""
        store.write(path, cur.replace(tin["old_str"], tin["new_str"], 1))
        return {"ok": True}
    if cmd == "insert":
        cur = (store.read(path) or "").splitlines()
        cur.insert(int(tin.get("insert_line", len(cur))), tin.get("insert_text", ""))
        store.write(path, "\n".join(cur) + "\n")
        return {"ok": True}
    if cmd == "delete":
        store.delete(path)
        return {"ok": True}
    if cmd == "rename":
        cur = store.read(path)
        if cur is not None:
            store.write(tin["new_path"], cur); store.delete(path)
        return {"ok": True}
    return {"error": f"unknown memory command: {cmd}"}
```
